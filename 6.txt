function getCartTotal() {
  const cart = loadCartFromStorage();

  // original total
  const total = cart.reduce((sum, item) => sum + Number(item.price), 0);

  // apply 10% discount if total > 1000
  if (total > 1000) {
    const discount = total * 0.10;
    return Math.round(total - discount);
  }

  return total;
}
